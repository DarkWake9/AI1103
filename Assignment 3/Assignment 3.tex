\section{\large GATE 2015 (EE paper 01 new 2), Q 27 (Electrical Engg section)}
Two players A, and B alternately keep rolling a fair dice. The person to get a six first wins the game. Given that player A starts the game, the probability that A wins the game is:\\[5pt]
\begin{enumerate}[label=(\Alph*)]
    \item $\dfrac{5}{11}$\\
    \item $\dfrac{1}{2}$ \\
    \item $\dfrac{7}{13}$\\
    \item $\dfrac{6}{11}$
\end{enumerate}
\section{\Large Solution}
\begin{itemize}
    \item Given the die is fair.
    \item So, for any given throw by A or B:\\
    The probability of getting 6 = $\dfrac{1}{6}$ = p (say)\\
    The probability of NOT getting 6 = $\dfrac{5}{6}$ = q (say)
\end{itemize}
Constraint: A starts the game and A should win.\\
$\implies$ Until A wins, both A and B cannot win.\\
%$\implies$ B(State 2) is absorbing state and it represents the state A loses and B wins.\\

    \begin{definition}
    Let $X_n$ be a random variable corresponding to the state of the game after $n^{th}$ throw of dice. $X_0$ is the initial state.\\
    \end{definition}
\begin{definition}
%\vspace{2cm}
    2 and 3 are ABSORBING states.\\
    Transition/Stochastic matrix is: T\\
    \centering
         \text{\small (From)} \begin{blockarray}{ cccc }
                \multicolumn{3}{c}{(\small To)}\\
                & 1 & 2 & 3 \\
                \begin{block}{ c @{\quad} [ @{\,} *{3}{c} @{\,} ] }
                    1 & $q^2$ & qp & p\\ 
                    2 & 0 & 1 & 0\\
                    3 & 0 & 0 & 1\\
                \end{block}
           \end{blockarray} = T
\end{definition}
\begin{theorem}
Every transition matrix can be partitioned as $\mleft(
    \begin{array}{c|c}
        Q & P\\
        \hline
        O & J\\
    \end{array}
    \mright)$\\
    $T^k$ approaches $\overline{T}$ as k increases.
    $\overline{T}$ is the limiting matrix and
    $\overline{T} = \myvec{O & NP\\O & J}$\\
        \end{theorem}
    \begin{tabular}{|c|l|}
    \hline
        Partition & Description\\
        \hline
        \multirow{2}{*}{Q} & Transition probability between\\&  non-absorbing states \\\hline
        \multirow{2}{*}{P} & Transition probabilities from non-\\
        & -absorbing state to absorbing states.\\\hline
        O & Null matrix\\
        \hline
        J & Identity matrix\\
        \hline
        N & Fundamental matrix\\
        \hline
    \end{tabular}
\begin{align}
    T = \mleft(
    \begin{array}{c| c c}
        q^2 & qp & p\\
        \hline
        0 & 1 & 0\\
        0 & 0 & 1\\
    \end{array}
    \mright)
    \end{align}
    \begin{table}[!htb]
    \caption{Matrices and corresponding values}
     \begin{tabular}{|c|c|c|}
        \hline
        Matrix & Description & Value\\\hline
        Q & \pr{X_{n+1} = 1|X_n = 1} & \myvec{q^2}  \\\hline
        P & \pr{X_{n+1} \neq 1|X_n = 1} & \myvec{qp & p}\\\hline
        N & $\mathlarger{\sum}_{n=0}^{\infty} Q^n = (I-Q)^{-1}$ & N = \myvec{\dfrac{1}{1-q^2}}\\\hline
    \end{tabular}
    \end{table}
    \begin{theorem}
    The matrix D = NP gives the probability of ending up in the absorbing states (2 and 3) when the chain starts from a non-absorbent state 1.\\
    By definition:
    \begin{equation*}
     D = \myvec{\mathlarger{\sum}_{n=1}^{\infty} \pr{\dfrac{X_n = 2}{X_0 = 1}} & \mathlarger{\sum}_{n=1}^{\infty} \pr{\dfrac{X_n = 3}{ X_0 = 1}}}
    \end{equation*}
\end{theorem}
    \begin{align}
        %D = \myvec{\pr{\dfrac{X_{n \to \infty} = 2}{X_0 = 1}} & \pr{\dfrac{X_{n \to \infty} = 3}{ X_0 = 1}}}\\
        D = NP = \myvec{\dfrac{qp}{1 - q^2} & \dfrac{p}{1 - q^2}}\\
        \pr{\text{A wins}} = D_2 = \dfrac{p}{1 - q^2} = \frac{\frac{1}{6}}{1 - \frac{25}{36}} = \dfrac{6}{11}
    \end{align}
